 1. # 1️Use an official Java runtime as the base image
 2. FROM openjdk:17-jdk-slim
 3.  
 4. # 2️Set the working directory inside the container
 5. WORKDIR /app
 6.  
 7. # 3️Copy the built JAR file into the container
 8. COPY target/violetApple-docker.jar app.jar
 9.  
10. # 4️Expose the application port
11. EXPOSE 8080
12.  
13. # 5️Run the application
14. ENTRYPOINT ["java", "-jar", "app.jar"]
